<div align="center">
  <h3 style="text-align: center">Building from source</h3>
</div>

<div align="center">
  <br/>
  <code>git clone https://github.com/theoteno/leko.git</code>
  <br/>
  <code>cd leko</code>
  <br/>
  <code>meson build --prefix=/usr</code>
  <br/>
  <code>cd build</code>
  <br/>
  <code>ninja && sudo ninja install</code>
  <br/>
</div>
